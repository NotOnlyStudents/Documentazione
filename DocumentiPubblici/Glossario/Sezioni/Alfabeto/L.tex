\TermineGlossario{Locale (ambiente)}
\DefinizioneGlossario{
	Ambiente di sviluppo, si intende la macchina di sviluppo locale di ogni sviluppatore.
}
\TermineGlossario{Leaflet}
\DefinizioneGlossario{
	 Libreria \glo{\textit{JavaScript}} per sviluppare mappe geografiche interattive. Supporta la maggior parte dei browser e degli standard \glo{\textit{HTML5}} e \glo{\textit{CSS}3}.
}
\TermineGlossario{Licenza MIT}
\DefinizioneGlossario{

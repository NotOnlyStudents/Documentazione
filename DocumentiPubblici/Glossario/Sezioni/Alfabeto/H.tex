\section{H}
\TermineGlossario{HTML/HTML5}
\DefinizioneGlossario{
	Linguaggio di markup per la strutturazione delle pagine web.	
}
\TermineGlossario{HTTP}
\DefinizioneGlossario{
	Protocollo di comunicazione dello strato Applicativo pensato per trasferire dati relativi alle pagine web.
}
\TermineGlossario{HTTPS}
\DefinizioneGlossario{
	Evoluzione di \glo{\textit{HTTP}} che fornisce funzionalità di cifratura e autenticità dei dati scambiati.
}

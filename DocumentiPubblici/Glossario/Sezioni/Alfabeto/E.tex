\section{E}
\TermineGlossario{E-commerce}
\DefinizioneGlossario{
	Piattaforma web attraverso la quale commercianti e acquirenti vengono in contatto.
}
\TermineGlossario{Efficace/Efficacia}
\DefinizioneGlossario{
	Grado di conformità del prodotto; misura la capacità di raggiungere gli obiettivi prefissati.
}
\TermineGlossario{Efficiente}
\DefinizioneGlossario{
	Misura l'abilità di raggiungere gli obiettivi impiegando il minimo numero di risorse.
}
